schema
	users
		uuid
		username
		password

	recipies
		uuid
		title
		steps
		fk user -> users

	ingredients
		uuid
		name
		metric
		quantity
		fk recipe -> recipies

UI endpoints
/        [feed page]
/user    [user info and configuration]
/login   [user login and sign up]
/recipe  [recipe details]

API endpoints

GET
/api/get/users                   [users dump]
/api/get/user/<id>               [user details]
/api/get/user/<id>/recipies      [user recipe dump]
/api/get/recipes                 [recipies dump]
/api/get/recipe/<id>             [recipe details]
/api/get/recipe/<id>/ingredients [recipe ingredients]

POST
/api/post/user					 [create a user]
  params:
  	username: string
	password: string
/api/post/recipe				 [create a recipe]
  params:
  	title: string,
	instructions: string,
	ingredients: Array<[]>
